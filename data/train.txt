from pathlib import Path

data_dir = Path("GenesisLLM/data")
data_dir.mkdir(parents=True, exist_ok=True)

text = """
대한민국의 수도는 서울이다.

파이썬은 범용 프로그래밍 언어이다.

Qwen은 대규모 언어 모델이다.

Transformer는 Self-Attention을 사용한다.

인공지능은 데이터를 학습하여 패턴을 찾는다.
"""

(data_dir / "train.txt").write_text(
    text,
    encoding="utf-8"
)

print("생성 완료!")
